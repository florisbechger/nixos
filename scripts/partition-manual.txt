# Change to root:
sudo -i
lsblk


#### MBR ####

## Set disk ##
DISK=/dev/vda

## Wipe disk ##
wipefs -a "$DISK"  

## Erase disk (takes a while) ##
#shred "$DISK" --iterations=1 --zero --verbose

## Partitioning MBR ##
parted "$DISK" -- mklabel msdos # MBR partition table
parted "$DISK" -- mkpart primary fat32 1MiB 512MiB
parted "$DISK" -- mkpart primary 512MiB 100%
parted "$DISK" -- set 1 boot on
fdisk -l

## Setup LVM MBR ##
pvcreate "$DISK"2
vgcreate vgsys "$DISK"2
lvcreate -n root vgsys -L 200GiB
lvcreate -n data vgsys -l 100%FREE

## Formatting MBR ##
mkfs.vfat -n BOOT "$DISK"1 -F 32
mkfs.ext4 -L root /dev/vgsys/root
mkfs.ext4 -L data /dev/vgsys/data

## Mounting MBR ##
mount /dev/vgsys/root /mnt
mkdir /data
mount /dev/vgsys/data /data
mkdir /mnt/boot
mount "$DISK"1 /mnt/boot


#### UEFI ####

## Set disk ##
DISK=/dev/nvme0n1

## Wipe disk ##
wipefs -a "$DISK"

## Erase disk (takes a while) ##
#shred "$DISK" --iterations=1 --zero --verbose

## Partitioning GPT ##
parted "$DISK" -- mklabel gpt
parted "$DISK" -- mkpart ESP fat32 1MiB 512MiB
parted "$DISK" -- set 1 esp on
parted "$DISK" -- mkpart primary 512MiB 100%
fdisk -l

## Setup LVM GPT ##
pvcreate "$DISK"p2
vgcreate vgsys "$DISK"p2
lvcreate -n root vgsys -L 160GiB
lvcreate -n data vgsys -l 100%FREE

## Formatting GPT ##
mkfs.vfat -n BOOT "$DISK"p1 -F 32
mkfs.ext4 -L root /dev/vgsys/root
mkfs.ext4 -L data /dev/vgsys/data

## Mounting GPT ##
mount /dev/vgsys/root /mnt
mkdir /data
mount /dev/vgsys/data /data
mkdir /mnt/boot
mount "$DISK"p1 /mnt/boot

## SETUP ##

## Configuration ##
nixos-generate-config --root /mnt
cd /mnt/etc/nixos

## Reference ##
https://nix.dev/manual/nix/2.28/installation/multi-user

nixos-install
